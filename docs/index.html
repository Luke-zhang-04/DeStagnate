State example

                
// Browser env, everything exposed to one global var, or namespace
const DS = DeStagnate,
    createChild = DS.createChild.default

class Counter extends DS.DeStagnate {

    constructor (props) {
        super(props)

        this.state = {
            count: 0,
        }
    }

    componentDidMount = () => {
        setInterval(() => {
            this.setState({count: this.state.count + 1})
        }, 1000)
    }

    render = () => createChild("div", {}, [
        createChild("p", {}, "Seconds:"),
        createChild("p", {}, this.state.count)
    ])

}

const counter = new Counter(document.querySelector("#counter"))

counter.mount() // Must call once

Calculator example

                
// Browser env, everything exposed to one global var, or namespace
const DS = DeStagnate,
    createChild = DS.createChild.default

class Calculator extends DS.DeStagnate {

    constructor (props) {
        super(props)

        this.state = {
            calculation: "",
        }
    }

    /**
     * Create a calculator button
     * @param {string} text - text of button
     * @param {string | undefined} append - value to append to
     * calculaton
     * @returns {HTMLElement} col element
     */
    _calcButton = (text, append) => createChild("div", {
        class: "col-3 calc-btn", // Bootstrap grid
        onClick: () => this.setState({
            calculation: this.state.calculation + (
                append ? append : text
            ),
        }),
    }, text)

    /**
     * Evaluate current claculation
     * @returns {number} clculation
     */
    _evalCalc = () => eval(this.state.calculation)

    /**
     * Calculator buttons for 3 numbered rows
     * @type {Array.}
     */
    _numBtns = [
        createChild("div", {class: "calc-btns row"}, 
            ["7", "8", "9", ["x", "*"]].map((val) => (
                this._calcButton(
                    val instanceof Array ? val[0] : val,
                    val instanceof Array ? val[1] : val,
                )
            ))
        ),
        createChild("div", {class: "calc-btns row"}, 
            ["4", "5", "6", "-"].map((val) => (
                this._calcButton(
                    val instanceof Array ? val[0] : val,
                    val instanceof Array ? val[1] : val,
                )
            ))
        ),
        createChild("div", {class: "calc-btns row"}, 
            ["1", "2", "3", "+"].map((val) => (
                this._calcButton(
                    val instanceof Array ? val[0] : val,
                    val instanceof Array ? val[1] : val,
                )
            ))
        ),
    ]

    /**
     * Calculator Display
     * @returns {HTMLElement} display
     */
    _calcDisplay = () => createChild(
        "div",
        {class: "calc-display"},
        this.state.calculation,
    )

    render = () => createChild("div", {}, [
        this._calcDisplay(),
        createChild("div", {class: "calc-btns row"}, [
            createChild("div", {
                class: "col-3 calc-btn clear",
                onClick: () => this.setState({calculation: ""}),
            }, "C"), // Clear
            createChild("div", {
                class: "col-3 calc-btn",
                onClick: () => this.setState({
                    calculation: this.state.calculation.slice(
                        0, this.state.calculation.length - 1,
                    ),
                }),
            }, "\u2190"), // Backspace

            this._calcButton("%"),
            this._calcButton("\u00f7", "/"),
        ]),
        ...this._numBtns,
        createChild("div", {class: "calc-btns row"}, [
            this._calcButton("0"),
            this._calcButton("."),
            createChild("div", {
                class: "col-6 calc-btn equals",
                onClick: () => this.setState({
                    calculation: this._evalCalc(),
                }),
            }, "="), // Equals
        ]),
    ])

}

const calculator = new Calculator(
    document.querySelector("#calculator")
)

calculator.mount() // Must call once
