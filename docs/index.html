// Const declarations since these variables are globally defined
declare type DeStagnate = typeof import("../lib")
declare const DeStagnate: DeStagnate

(() => {
    const {createElement} = DeStagnate,
        badgeUrl = "https://img.shields.io/github/size/luke-zhang-04/DeStagnate/dist/deStagnate.bundle.min.js?label=deStagnate.bundle.min.js&style=for-the-badge"

    /**
     * Create nav item
     * @param {string} file - file to link to
     * @param {string} content - content of item
     * @returns {HTMLElement} nav item
     */
    const createNavItem = (file: string, content: string): HTMLElement => createElement("li", null,
            createElement("a", {class: "text-dark", href: file}, content)
        ),
        /**
         * All pages with their location and display name
         */
        pages = [
            ["index.html", "Home"],
            ["state.html", "State example"],
            ["props.html", "Props example"],
            ["eventListener.html", "Event listener example"],
            ["ref.html", "Ref example"],
            ["calculator.html", "Calculator Example"],
            ["namespace.html", "Namespaced element (SVG) example"],
            ["ticTacToe.html", "TicTacToe example"],
            ["ticTacToe.html", "Nested component example"],
        ]

    /**
      * Append to elements to nav
      */
    document.getElementById("nav").appendChild(
        createElement("div", {class: "p-4 bg-light"},
            createElement("div", {class: "row"},
                createElement("div", {class: "col-12 col-sm-6"}, 
                    createElement(
                        "a",
                        {
                            href: "https://github.com/Luke-zhang-04/DeStagnate/blob/master/dist/deStagnate.bundle.min.js",
                            target: "_blank",
                            rel: "noopener noreferrer",
                            class: "pl-4"
                        },
                        createElement(
                            "img",
                            {
                                src: badgeUrl,
                                alt: "bundle min size",
                                class: "mb-3",
                            },
                        )
                    ),
                    createElement("ul", null,
                        pages.map((val) => createNavItem(val[0], val[1])),
                    ),
                ),
                createElement("div", {class: "col-12 col-sm-6 col-lg-4"},
                    createElement("a", {href: "index.html"},
                        createElement("img", {src: "./logo.svg", class: "w-100"}),
                    )
                ),
            ),
        ),
    )
})()

DeStagnate vs Vanilla DOM

Here's a side-by-side comparison of DeStagnate vs Vanilla DOM Manipulation

                    
/**
 * DeStagnate
 */
const {createElement} = DeStagnate,
    class = "container bg-light text-dark py-1 br-1 my-1"

document.getElementById("compare").children[0].appendChild(
    createElement("div", {class},
        createElement("p", {class: "m-0 p-0"}, "Hello World!")
    )
)
                    
                
                    
/**
 * Vanilla DOM
 */
const helloWorldContainer = document.createElement("div"),
    helloWorld = document.createElement("p")

helloWorldContainer.classList.add("container")
helloWorldContainer.classList.add("bg-light")
helloWorldContainer.classList.add("text-dark")
helloWorldContainer.classList.add("py-1")
helloWorldContainer.classList.add("my-1")
helloWorldContainer.classList.add("br-1")
helloWorld.classList.add("m-0")
helloWorld.classList.add("p-0")
helloWorld.innerText = "Hello World!"

helloWorldContainer.appendChild(helloWorld)

document.getElementById("compare")
    .children[1]
    .appendChild(helloWorldContainer)
