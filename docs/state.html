State example

This example was written with JSX, which needs to be transpiled. See using JSX.

Source Code 
                
import * as DeStagnate from "destagnate"

class Counter extends DeStagnate.Component {

    constructor (parent) {
        super(parent)

        this.state = {
            count: 0,
        }
    }

    componentDidMount = () => {
        setInterval(() => {
            this.setState({count: this.state.count + 1})
        }, 1000)
    }

    render = () => <div>
        <p>Seconds:</p>
        <p>{this.state.count}</p>
    </div>

}

const counter = new Counter(document.querySelector("#counter"))

counter.mount() // Must call once
